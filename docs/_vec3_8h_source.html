437  CANNON_FORCE_INLINE real_t dot(const Vec3& v) const {
438 #ifdef CANNON_NEON_64
439  float32x4_t vd = vmulq_f32(mVec128, v.mVec128);
440  float32x2_t x = vpadd_f32(vget_low_f32(vd), vget_low_f32(vd));
441  x = vadd_f32(x, vget_high_f32(vd));
442  return vget_lane_f32(x, 0);
443 #else
444  return x * v.x + y * v.y + z * v.z;
445 #endif
