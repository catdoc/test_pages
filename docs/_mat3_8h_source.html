125  CANNON_FORCE_INLINE Vec3& vmult(const Vec3& v, Vec3& target) const {
126 #ifdef CANNON_NEON_64
127  const float32x4_t vv = v.mVec128;
128  const float32x2_t vlo = vget_low_f32(vv);
129  const float32x2_t vhi = vget_high_f32(vv);
130 
131  float32x4_t c0, c1, c2;
132 
133  c0 = (float32x4_t)vandq_s32((int32x4_t)m[0].mVec128, b3vFFF0Mask);
134  c1 = (float32x4_t)vandq_s32((int32x4_t)m[1].mVec128, b3vFFF0Mask);
135  c2 = (float32x4_t)vandq_s32((int32x4_t)m[2].mVec128, b3vFFF0Mask);
136 
137  c0 = vmulq_lane_f32(c0, vlo, 0);
138  c1 = vmulq_lane_f32(c1, vlo, 1);
139  c2 = vmulq_lane_f32(c2, vhi, 0);
140  c0 = vaddq_f32(c0, c1);
141  c0 = vaddq_f32(c0, c2);
142 
143  target.set128(c0);
144 #else
145  target.x = m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z;
146  target.y = m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z;
147  target.z = m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z;
148 #endif
149  return target;
150  }
