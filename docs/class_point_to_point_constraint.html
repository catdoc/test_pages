Detailed Description

Connects two bodies at given offset points.

Example:

using namespace cannon;
Body::BodyOptions bodyOptions(world);
bodyOptions.id = world->genBodyId();
bodyOptions.mass = 1;
auto bodyA = Body::create(bodyOptions);
bodyOptions.id = world->genBodyId();
auto bodyB = Body::create(bodyOptions);
bodyA->get_position().set(-1, 0, 0);
bodyB->get_position().set(1, 0, 0);
bodyA->addShape(shapeA);
bodyB->addShape(shapeB);
world->addBody(bodyA);
world->addBody(bodyB);
Vec3 localPivotA = Vec3(1, 0, 0);
Vec3 localPivotB = Vec3(-1, 0, 0);
auto constraint = new PointToPointConstraint(world->genConstraintId(), bodyA, bodyB, localPivotA, localPivotB);
world->addConstraint(constraint);
