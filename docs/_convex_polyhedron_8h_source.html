286  void set_faces(const int* faces, int faceCount) {
287  int i, j;
288  int facePointCount;
289  _faces.clear();
290  //
291  for (i = 0; i < faceCount; i += facePointCount + 1) {
292  // The facePointCount of current face
293  facePointCount = faces[i];
294 
295  std::vector<int> face;
296  for (j = 1; j <= facePointCount; ++j) {
297  face.emplace_back(faces[i + j]);
298  }
299  _faces.emplace_back(face);
300  }
301  }
