205  CANNON_FORCE_INLINE Quaternion& mult(const Quaternion& q, Quaternion& target) const {
206 #ifdef CANNON_NEON_64
207  float32x4_t vQ1 = mVec128;
208  float32x4_t vQ2 = q.get128();
209  float32x4_t A0, A1, B1, A2, B2, A3, B3;
210  float32x2_t vQ1zx, vQ2wx, vQ1yz, vQ2zx, vQ2yz, vQ2xz;
211 
212  {
213  float32x2x2_t tmp;
214  tmp = vtrn_f32(vget_high_f32(vQ1), vget_low_f32(vQ1)); // {z x}, {w y}
215  vQ1zx = tmp.val[0];
216 
217  tmp = vtrn_f32(vget_high_f32(vQ2), vget_low_f32(vQ2)); // {z x}, {w y}
218  vQ2zx = tmp.val[0];
219  }
220  vQ2wx = vext_f32(vget_high_f32(vQ2), vget_low_f32(vQ2), 1);
221 
222  vQ1yz = vext_f32(vget_low_f32(vQ1), vget_high_f32(vQ1), 1);
223 
224  vQ2yz = vext_f32(vget_low_f32(vQ2), vget_high_f32(vQ2), 1);
225  vQ2xz = vext_f32(vQ2zx, vQ2zx, 1);
226 
227  A1 = vcombine_f32(vget_low_f32(vQ1), vQ1zx); // X Y z x
228  B1 = vcombine_f32(vdup_lane_f32(vget_high_f32(vQ2), 1), vQ2wx); // W W W X
229 
230  A2 = vcombine_f32(vQ1yz, vget_low_f32(vQ1));
231  B2 = vcombine_f32(vQ2zx, vdup_lane_f32(vget_low_f32(vQ2), 1));
232 
233  A3 = vcombine_f32(vQ1zx, vQ1yz); // Z X Y Z
234  B3 = vcombine_f32(vQ2yz, vQ2xz); // Y Z x z
235 
236  A1 = vmulq_f32(A1, B1);
237  A2 = vmulq_f32(A2, B2);
238  A3 = vmulq_f32(A3, B3); // A3 *= B3
239  A0 = vmulq_lane_f32(vQ2, vget_high_f32(vQ1), 1); // A0 * B0
240 
241  A1 = vaddq_f32(A1, A2); // AB12 = AB1 + AB2
242  A0 = vsubq_f32(A0, A3); // AB03 = AB0 - AB3
243 
244  // change the sign of the last element
245  A1 = (float32x4_t)veorq_s32((int32x4_t)A1, (int32x4_t)b3vPPPM);
246  A0 = vaddq_f32(A0, A1); // AB03 + AB12
247 
248  target.set128(A0);
249 #else
250  real_t ax = this->x, ay = this->y, az = this->z, aw = this->w;
251  real_t bx = q.x, by = q.y, bz = q.z, bw = q.w;
252 
253  target.x = ax * bw + aw * bx + ay * bz - az * by;
254  target.y = ay * bw + aw * by + az * bx - ax * bz;
255  target.z = az * bw + aw * bz + ax * by - ay * bx;
256  target.w = aw * bw - ax * bx - ay * by - az * bz;
257 #endif
258  return target;
259  }

307  this->conjugate(target);

310 #ifdef CANNON_NEON_64
