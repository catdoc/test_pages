Detailed Description

Heightfield shape class. Height data is given as an array. These data points are spread out evenly with a given distance.

Example:

using namespace cannon;
std::vector<std::vector<float>> matrix;
int sizeX = 15, sizeY = 15;
for (int i = 0; i < sizeX; i++) {
std::vector<float> row;
for (int j = 0; j < sizeY; j++) {
float height = std::cos(i/(float)sizeX * CANNON_PI * 2)* std::cos(j/(float)sizeY * CANNON_PI * 2) + 2;
if(i==0 || i == sizeX-1 || j==0 || j == sizeY-1) {
height = 3;
}
row.emplace_back(height);
}
matrix.emplace_back(row);
}
cannon::Body::BodyOptions hfBodyOptions(world, world->genBodyId());
hfBodyOptions.mass = 0;
auto hfBody = Body::create(hfBodyOptions);
auto hfShape = new Heightfield(world->genShapeId(), world, matrix, 0, 0, 1);
hfBody->addShape(hfShape);
hfBody->get_position().set(-sizeX * hfShape->get_elementSize() / 2.0, -20, -10);
world->addBody(hfBody);
