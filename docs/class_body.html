Detailed Description

A RigidBody in the physics world.

Example:

using namespace cannon;
cannon::Body::BodyOptions bodyOptions(world);
bodyOptions.id = world->genBodyId();
bodyOptions.mass = 1;
auto body = Body::create(bodyOptions);
auto shape = Sphere::create(world->genShapeId(), 1);
body->addShape(shape);
world->addBody(body);
